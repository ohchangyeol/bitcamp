data type =  byte(1byte), short(2byte), int(4byte) , long(8byte) , float(4byte) , double(8byte) , char(2byte) , boolean(1byte)

casting = 형변환
	-묵시적 형변환
	 - 작은 데이터는 큰 데이터로 자동변환
	 - int 보다 작은 값을 연산할땐 무조건 int로 반환
	 sData + bData = bData
 
	-명시적 형변환
	 - 작은 데이터를 큰 데이터로 강제로 명시함.
	 sData + bData = (small)sData
ex)
byte a = 1;
byte b = 2;
int c = a + b ; -> literal값 = int 3;
int d = a + c ; -> literal값 = int 4;

byte c = a + b ; -> 오류 - 그릇이 다름.

byte c = (byte) a + b ;  literal값 = byte 3;
